FONCTION calculerMeilleureNote(notes)

	FOR note dans notes 
		SI meilleure_note < note
			meilleure_note <- note
		FIN SI
	FIN FOR
	RETOURNER meilleure_note
FIN

FONCTION calculerAdmis(notes)

	FOR note dans notes
		SI note >= 10
			eleve_admis <- eleve_admis + 1
		FIN SI
	FIN FOR
	RETOURNER eleve_admis
FIN
