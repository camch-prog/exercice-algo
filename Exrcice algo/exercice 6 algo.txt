ex 1 : 

erreur : SI age > 18 

correction:
age <- 16 
	SI age >= 18
		Afficher "Majeur"
	SINON
		Afficher "Mineur"
	FIN SI

ex 2 :

erreur : somme <- note

correction:

notes <- [10,12,14]
somme <- 0 
	POUR chaque note dans notes
		somme <- somme + note

	FIN POUR
moyenne <- somme/3
Afficher moyenne
